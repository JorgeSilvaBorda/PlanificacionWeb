 QUERY para obtener los datos ordenados y generar el JSON:
 
 SELECT
	A.FECHA,
	A.IDRECURSO,
	A.NOMBRE,
	CASE WHEN B.PORCENTAJE IS NULL THEN 0 ELSE B.PORCENTAJE END AS PORCENTAJE
FROM

(SELECT DISTINCT FECHA, RECURSO.ID AS IDRECURSO, NOMBRE FROM(SELECT FECHAINI AS FECHA FROM ETAPAPROYECTO UNION ALL SELECT FECHAFIN AS FECHA FROM ETAPAPROYECTO), RECURSO) A 

LEFT JOIN 

(SELECT
	A.FECHAINI AS FECHA, 
	C.ID AS IDRECURSO,
	C.NOMBRE,
	B.PORCENTAJE
FROM
	ETAPAPROYECTO A INNER JOIN ETAPAPROYECTOPERSONAL B
	ON A.ID = B.IDETAPAPROYECTO INNER JOIN RECURSO C
	ON B.IDRECURSO = C.ID 
UNION ALL
SELECT
	A.FECHAFIN AS FECHA, 
	C.ID AS IDRECURSO,
	C.NOMBRE,
	B.PORCENTAJE
FROM
	ETAPAPROYECTO A INNER JOIN ETAPAPROYECTOPERSONAL B
	ON A.ID = B.IDETAPAPROYECTO INNER JOIN RECURSO C
	ON B.IDRECURSO = C.ID 
ORDER BY
	1 ASC,
	3 ASC) B
ON A.FECHA = B.FECHA
AND A.IDRECURSO = B.IDRECURSO

ORDER BY A.FECHA ASC, A.NOMBRE ASC
